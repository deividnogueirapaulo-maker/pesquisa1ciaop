function atualizarEscala() { const agora = new Date(); // Definimos uma data de referência onde o 1º Pelotão entrou às 08:00h // (Se 28/04 é o 2º Pelotão, então 27/04 foi o 1º Pelotão) // Nota: No JS, o mês começa em 0 (Janeiro = 0, Abril = 3) const dataReferencia = new Date(2026, 3, 27); // Criamos uma cópia da data atual zerando as horas para evitar cálculos quebrados let dataOperacional = new Date(agora.getFullYear(), agora.getMonth(), agora.getDate()); // Lógica da Mudança às 08:00: // Se ainda não deu 8 da manhã, para fins de escala, ainda é o "dia anterior" if (agora.getHours() < 8) { dataOperacional.setDate(dataOperacional.getDate() - 1); } // Calculamos a diferença de dias inteiros const diffTempo = dataOperacional.getTime() - dataReferencia.getTime(); const diffDias = Math.round(diffTempo / (1000 * 60 * 60 * 24)); // Cálculo do Pelotão (Sequência: 1, 2, 3, 4) let pelotao = (diffDias % 4) + 1; // Ajuste para números negativos (caso alguém acesse com data retroativa no PC) if (pelotao <= 0) pelotao += 4; const dataTexto = dataOperacional.toLocaleDateString('pt-BR', { year: 'numeric', month: 'long', day: 'numeric' }); document.getElementById('data-pelotao').innerText = `Data: ${dataTexto} - ${pelotao}º Pelotão Operacional`; }
